pub struct Ping {
    #[prost(string, tag = "1")]
    pub mu: ::prost::alloc::string::String,
}
#[derive(Clone, PartialEq, ::prost::Message)]
pub struct Pong {
    #[prost(string, tag = "1")]
    pub mu: ::prost::alloc::string::String,
}
#[doc = r" Generated client implementations."]
#[allow(dead_code, unused_imports)]
pub mod mu_client {
    use prost::Message;
    use std::ops::{Deref, DerefMut};
    #[derive(Debug, Clone)]
    pub struct MuClient {
        inner: hrpc::Client,
    }
    impl MuClient {
        pub fn new(
            inner: hrpc::reqwest::Client,
            server: hrpc::url::Url,
        ) -> hrpc::ClientResult<Self> {
            Ok(Self {
                inner: hrpc::Client::new(inner, server)?,
            })
        }
        pub async fn mu(
            &mut self,
            request: impl Into<super::Ping>,
        ) -> hrpc::ClientResult<super::Pong> {
            let req = self.inner.make_request(request.into(), "/test.Mu/Mu")?;
            self.inner.execute_request(req).await
        }
        pub async fn mu_mute(
            &mut self,
        ) -> hrpc::ClientResult<hrpc::Socket<super::Ping, super::Pong>> {
            self.connect_socket("/test.Mu/MuMute").await
        }
    }
    impl DerefMut for MuClient {
        fn deref_mut(&mut self) -> &mut Self::Target {
            &mut self.inner
        }
    }
    impl Deref for MuClient {
        type Target = hrpc::Client;
        fn deref(&self) -> &Self::Target {
            &self.inner
        }
    }
}
